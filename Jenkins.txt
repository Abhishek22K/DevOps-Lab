JENKINSFILE (DOCKER BUILD & RUN)

pipeline {
    agent any

    environment {
        DOCKER_IMAGE = 'hello-world-python:latest'
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/...'
            }
        }

        stage('Docker Build') {
            steps {
                sh "docker build -t ${DOCKER_IMAGE} ."
            }
        }

        stage('Docker Run') {
            steps {
                sh "docker run --rm ${DOCKER_IMAGE}"
            }
        }
    }
}

FULL CI/CD PIPELINE (NGINX)

pipeline {
    agent any

    environment {
        DOCKER_IMAGE = 'mynginx/webapp'
    }

    stages {
        stage('Clone') {
            steps {
                git url: 'https://github.com/...'
            }
        }

        stage('Build') {
            steps {
                sh 'docker build -t $DOCKER_IMAGE .'
            }
        }

        stage('Deploy') {
            steps {
                sh 'docker rm -f webapp-container || true'
                sh 'docker run -d -p 80:80 --name webapp-container $DOCKER_IMAGE'
            }
        }
    }
}
