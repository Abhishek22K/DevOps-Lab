---

Real Example

docker run -d --name mysql-db -e MYSQL_ROOT_PASSWORD=root -e MYSQL_DATABASE=school -p 3306:3306 mysql:8
docker build -t student-app .
docker run --name app student-app
docker stop app
docker stop mysql-db
docker container prune

---

With Custom Network

docker network create mynet
docker run -d --network mynet --name mysql-db -e MYSQL_ROOT_PASSWORD=root -e MYSQL_DATABASE=school mysql:8
docker run --network mynet --name app student-app
docker exec -it mysql-db mysql -uroot -proot

Without Volume (Data Loss Example)

NGINX WITH VOLUME

mkdir webpages
echo "Hello" > webpages/index.html

# Windows
docker run -p 8080:80 --name another_nginx -v %cd%/webpages:/usr/share/nginx/html -d nginx

# Linux
docker run -p 8080:80 --name another_nginx -v ${PWD}/webpages:/usr/share/nginx/html -d nginx

docker exec -it another_nginx /bin/bash
cat /usr/share/nginx/html/index.html
