INSTALLATION (MINIKUBE)

sudo install minikube-linux-amd64 /usr/local/bin/minikube

minikube kubectl -- get nodes

source ~/.bashrc

CLEANUP SCRIPT (AWS)

for cluster in $(aws eks list-clusters --region us-east-1 --query "clusters[]" --output text); do
  for nodegroup in $(aws eks list-nodegroups --cluster-name $cluster --region us-east-1 --query "nodegroups[]" --output text); do
    aws eks delete-nodegroup --cluster-name $cluster --nodegroup-name $nodegroup --region us-east-1
  done
  aws eks delete-cluster --name $cluster --region us-east-1
done
